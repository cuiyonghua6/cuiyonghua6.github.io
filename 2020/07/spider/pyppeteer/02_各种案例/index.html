    import asyncio
from pyppeteer import launch


async def main(url):
    browser = await launch()
    page = await browser.newPage()
    res = await page.goto(url, options={'timeout': 30000})

    data = await page.content()
    title = await page.title()
    resp_cookies = await page.cookies()  # cookie
    resp_headers = res.headers  # 响应头
    resp_status = res.status  # 响应状态

    print(data)
    print(title)
    print(resp_headers)
    print(resp_status)


if __name__ == '__main__':
    url = "http://www.nows.fun/"
    asyncio.get_event_loop().run_until_complete(main(url))

    import asyncio
from pathlib import Path

from pyppeteer import launch


async def main(url):
    browser = await launch()
    page = await  browser.newPage()

    await page.setUserAgent('Mozilla/5.0 (Windows NT 6.1; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/70.0.3538.67 Safari/537.36')

    # 是否启用JS，enabled设为False，则无渲染效果
    await page.setJavaScriptEnabled(enabled=True)

    # 注入 js 文件
    CURDIR = Path(__file__).parent
    JS_AJAX_HOOK_LIB = str(CURDIR / 'static' / 'ajaxhook.min.js')
    await page.addScriptTag(path=JS_AJAX_HOOK_LIB)

    res = await page.goto(url, waitUntil=["networkidle0", "load", "domcontentloaded"],options={'timeout': 0})

    # cookie
    cookies = await page.cookies()

    # 网站 title
    title = await page.title()

    # html 内容
    contents = await page.content()

    # 响应头
    res_headers = res.headers

    # 响应状态
    res_status = res.status

    # 等待
    await asyncio.sleep(2)

async def goto(page, url):
    '''请求加载是否完成，无网都需要处理'''
    while True:
        try:
            await page.goto(url, {
                'timeout': 0,
                'waitUntil': 'networkidle0'
            })
            break
        except (pyppeteer.errors.NetworkError,
                pyppeteer.errors.PageError) as ex:
            # 无网络 'net::ERR_INTERNET_DISCONNECTED','net::ERR_TUNNEL_CONNECTION_FAILED'
            if 'net::' in str(ex):
                await asyncio.sleep(10)
            else:
                raise


async def request_check(self, req):
    '''filter requests
    Disable image loading
    '''
    if req.resourceType in ["image", "media", "eventsource", "websocket", "stylesheet", "font"]:
        await req.abort()
    elif "logout" in req.url or "delete" in req.url or "signout" in req.url:
        await req.abort()
    else:
        await req.continue_()

    browser = await launch()
    page = await browser.newPage()
    await page.setRequestInterception(True)
    page.on('request', self.request_check)
